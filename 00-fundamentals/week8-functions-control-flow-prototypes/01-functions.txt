funciones
  un tipo de objeto especial que permite almacenar un bloque de codigo

  las funciones son valores (no variables)
    asi que se pueden asignar a variables como cualquier otro valor

function expression
  ejemplo:

    let nombreFuncion = function() {
      // cuerpo de la función
    }

function declaration
  ejemplo:

    function nombreFuncion() {
      // cuerpo de la función
    }

incluso las function declaration aunque parezca que no son valores
  porque no se ve uso de 'let' o 'const' si crean una variable internamente

    ejemplo:

      function saludar() {
        console.log("¡Hola, mundo!");
      }

      saludar = "Otro valor";
      console.log(saludar); // "Otro valor"

scope (alcance)

  determina la visibilidad o accesibilidad de variables

  variables locales
    variables definidas dentro de una función son locales
    no pueden ser accedidas fuera de la función
      ejemplo
        function mostrarMensaje() {
          let mensaje = "Soy una variable local"; 
          console.log(mensaje);
        }

        mostrarMensaje(); // Soy una variable local
        console.log(mensaje); // Error: mensaje no está definido

  variables externas
    variables definidas fuera de la función
    pueden ser accedidas desde el interior de la función
      ejemplo
        let mensajeExterno = "Soy una variable externa";

        function mostrarMensajeExterno() {
          console.log(mensajeExterno);
        }

        mostrarMensajeExterno(); // Soy una variable externa

shadowing
  para el caso en el que se tengan mismos nombres de variables
    con variables externas y variables locales
    los valores de variables locales seran las utilizadas dentro de la funcion
      ejm
        let nombre = "Sebastian";

        function saludarJuan() {
          let nombre = "Juan";
          console.log("Hola" + " " + nombre);
        }

        saludarJuan(); // "Hola Juan"
        console.log(nombre); // "Sebastian"

parametros con valores por defecto
  ejemplo
    function crearSaludo(nombre = "desconocido") {
      return "Hola, " + nombre + "!";
    }

    console.log(crearSaludo()); // "Hola, desconocido!"
    console.log(crearSaludo("Andrea")); // "Hola, Andrea!"

convenciones al nombrar funciones

  claro y descriptivo
    el nombre de la función debe ser un reflejo de lo que hace: calcularPromedio(), obtenerFechaActual()

  usar camelcase
    la convención más común en javascript es el camelcase
      donde la primera palabra comienza en minúscula y las siguientes en mayúscula: obtenerDatosUsuario()
    es la misma convención que seguiremos para los nombres de variables y propiedades

  evitar nombres genéricos
    nombres como func1(), data(), ejecutar() no son descriptivos y dificultan la lectura del código
