funciones
  un tipo de objeto especial que permite almacenar un bloque de codigo

  las funciones son valores (no variables)
    asi que se pueden asignar a variables como cualquier otro valor

function expression
  ejemplo:

    let nombreFuncion = function() {
      // cuerpo de la función
    }

function declaration
  ejemplo:

    function nombreFuncion() {
      // cuerpo de la función
    }

incluso las function declaration aunque parezca que no son valores
  porque no se ve uso de 'let' o 'const' si crean una variable internamente

    ejemplo:

      function saludar() {
        console.log("¡Hola, mundo!");
      }

      saludar = "Otro valor";
      console.log(saludar); // "Otro valor"

scope (alcance)

  determina la visibilidad o accesibilidad de variables

  variables locales
    variables definidas dentro de una función son locales
    no pueden ser accedidas fuera de la función
      ejemplo
        function mostrarMensaje() {
          let mensaje = "Soy una variable local"; 
          console.log(mensaje);
        }

        mostrarMensaje(); // Soy una variable local
        console.log(mensaje); // Error: mensaje no está definido

  variables externas
    variables definidas fuera de la función
    pueden ser accedidas desde el interior de la función
      ejemplo
        let mensajeExterno = "Soy una variable externa";

        function mostrarMensajeExterno() {
          console.log(mensajeExterno);
        }

        mostrarMensajeExterno(); // Soy una variable externa
