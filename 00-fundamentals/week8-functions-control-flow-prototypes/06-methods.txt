las funciones son valores (como los strings y numbers)
  asi que se pueden asociar a una propiedad de un objeto

cuando usas console.log()
  tambien estas usando una funcion dentro de un objeto

se llama "método" a una propiedad de un objeto que apunta a una función
  es solo un nombre para un caso especifico. no hay nada en especial en el fondo
  es solo para poder diferenciar 'metodo' de 'funcion'

una funcion se puede llamar con su nombre directamente
  un metodo se llama a traves de su objeto

  ejemplo
    function saludar() {
      console.log("¡Hola!");
    }

    saludar(); // llamamos a la función por su nombre

    const persona = {
      nombre: "Luis",
      saludar: function() {
        console.log("¡Hola!");
      }
    };

    persona.saludar(); // llamamos al método a partir de su objeto

crear metodos
  definiendo la función directamente en el objeto

    const pokemon = {
      especie: "Pikachu",
      saludar: function() {
        console.log("Pikachu saluda");
      },
      // otra forma en ES6
      saludar2() {
        console.log("Pikachu saluda");
      }
    };

  referenciando una función existente

    function pikachuSaluda() {
      console.log("Pikachu saluda");
    }

    const pokemon = {
      especie: "Pikachu",
      saludar: pikachuSaluda,
    };

de que sirve colocar la función dentro del objeto? 
  al estar "encapsulados" en sus objetos, no "contaminan" el scope global de tu programa

  en vez de llenar el scope global con varios nombres de funciones
    function volarAvion() { /* implementación... */ }
    function volarAve() { /* implementación... */ }

  se usan metodos
    const ave = {
      // ...otras propiedades de ave
      volar() {
        /* implementación... */
      }
    }

    const avion = {
      // ...otras propiedades de avion
      volar() {
        /* implementación... */
      }
    }

definicion de objeto actualizada
  un objeto es un conjunto de propiedades y métodos relacionados
