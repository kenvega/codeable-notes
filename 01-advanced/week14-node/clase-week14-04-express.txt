install/setup express

creas carpeta de proyecto
  mkdir express-playground
  cd express-playground

instala paquetes
  npm init -y
  npm install express
  npm install -D typescript @types/node nodemon ts-node @types/express

crea archivo de tsconfig.json
  npx tsc --init

actualiza el tsconfig.json con esto
  {
    "compilerOptions": {
      /* Language and Environment */
      "target": "ESNext",

      /* Modules */
      "module": "NodeNext",
      "rootDir": "./src",

      /* Emit */
      "outDir": "./build",

      /* Interop Constraints */
      "esModuleInterop": true,
      "forceConsistentCasingInFileNames": true,

      /* Type Checking */
      "strict": true,
      "noImplicitThis": true,

      "noUnusedLocals": true,
      "noUnusedParameters": true,

      "noFallthroughCasesInSwitch": true,

      /* Completeness */
      "skipLibCheck": true
    }
  }

codigo inicial de app de express

  import express from "express";

  const app = express();
  const port = 3000;

  app.get("/", (req, res) => {
    console.log(`${req.method} ${req.url}`);
    res.send("Hola Mundo con Express y TypeScript!");
  });

  app.listen(port, () => {
    console.log(`Servidor corriendo en http://localhost:${port}`);
  });
