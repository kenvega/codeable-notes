const baseUrl = 'https://rickandmortyapi.com/api'; try { const response = await fetch(`${baseUrl}/character`); if (!response.ok) throw new Error(response.status); const charactersData = await response.json(); console.log(charactersData); const characters = charactersData.results; for (const character of characters) { const characterCard = createCharacterCard(character); document.body .querySelector('#characters-container') .appendChild(characterCard); } } catch (error) { console.log(error.message); } function createCharacterCard(character) { const card = document.createElement('div'); card.className = 'character-card'; card.innerHTML = `

${character.name}

Status: ${character.status}

Species: ${character.species}

Origin: ${character.origin.name}

Location: ${character.location.name}

`; return card; }
